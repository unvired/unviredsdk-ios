LoginParameters

Objective-C

@interface LoginParameters : NSObject

Swift

class LoginParameters : NSObject

This singleton instance serves as the input to the login API of AuthenticationService. The following code snippet sets some default values for the LoginParameters and calls the Login API.

  • (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions { self.window = [[UIWindow alloc] initWithFrame:[[UIScreen mainScreen] bounds]];

// Get the Demo Data NSString * filePath = [[NSBundle mainBundle] pathForResource: @“demo_data” ofType: @“xml” ]; NSData * demoData = [NSData dataWithContentsOfFile:filePath ];

// Set all the Required Parameters in LoginParams LoginParameters * loginParameters = [LoginParameters sharedParameters]; loginParameters.applicationWindow = self.window; loginParameters.url = @“http://192.168.98.226:8080/UNI”; loginParameters.applicationTitle = NSLocalizedString(@“Unvired Sales for SAP”, nil); loginParameters.applicationLogoForiPad = NULL; loginParameters.applicationLogoForiPhone = NULL; loginParameters.redirectURL = @“http://www.unvired.com”; loginParameters.loginActivityListener = self; loginParameters.demoModeListener = self; loginParameters.demoData = demoData; loginParameters.demoModeRequired = YES;

loginParameters.supportedLoginTypes = @[@(LoginTypeSAP), @(LoginTypeUnviredId), @(LoginTypeADS)];

// Call the Login API [AuthenticationService login]; }
